# Task 1

DROP TABLE IF EXISTS SampleTable;
DROP TABLE IF EXISTS RenamedTable;

CREATE TABLE IF NOT EXISTS SampleTable (
    id SERIAL PRIMARY KEY,
    name VARCHAR(200),
    age INTEGER
);

INSERT INTO SampleTable (name, age) VALUES
    ('John Doe', 25),
    ('Jane Doe', 30),
    ('Graham Mumford', 50);

-- Rename the table
ALTER TABLE SampleTable RENAME TO RenamedTable;

-- Add a new column
ALTER TABLE RenamedTable ADD COLUMN city VARCHAR(200);

-- Update values in the table
UPDATE RenamedTable SET city = 'New York' WHERE id = 1;
UPDATE RenamedTable SET city = 'London' WHERE id = 3;

DELETE FROM RenamedTable WHERE id = 2;

SELECT * FROM RenamedTable;

SELECT first_name AS "First Name", last_name AS "Last Name" FROM employees;
SELECT DISTINCT department_id FROM employees;
SELECT * FROM employees ORDER BY first_name DESC;
SELECT first_name, last_name, salary, salary * 0.12 AS PF FROM employees;

SELECT employee_id, first_name, last_name, ROUND(salary / 12, 2) AS monthly_salary FROM employees;
